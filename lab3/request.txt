SELECT * FROM `book` WHERE price is NULL; /*Вивести книги у яких не введена ціна або ціна дорівнює 0*/

SELECT * FROM `book` WHERE price is NOT NULL AND circulation is NULL; /*Вивести книги у яких введена ціна, але не введений тираж*/

SELECT * FROM `book` WHERE datee is NULL; /*Вивести книги, про дату видання яких нічого не відомо.*/

SELECT * FROM `book` WHERE (CURRENT_DATE - datee)< (1-1-1); /*Вивести книги, з дня видання яких пройшло не більше року.*/

SELECT * FROM `book` ORDER BY price ASC; /*Вивести список книг-новинок, відсортованих за зростанням ціни*/

SELECT * FROM `book` WHERE (sides>300 AND sides <400) ORDER BY name DESC; /*Вивести список книг з числом сторінок від 300 до 400, відсортованих в зворотному алфавітному порядку назв*/

SELECT * FROM `book` WHERE (price > 20 AND price < 40) ORDER BY datee DESC;  /*Вивести список книг з ціною від 20 до 40, відсортованих за спаданням дати*/

SELECT * FROM `book` WHERE (price / sides) < 0.10;  /*Вивести книги, у яких ціна однієї сторінки < 10 копійок.*/

SELECT LENGTH(name) AS number FROM `book`;/*Вивести значення наступних колонок: число символів в назві*/
SELECT LEFT(name,20) AS FIRST_20 FROM `book`; /* перші 20 символів назви великими літерами*/

SELECT CONCAT(LEFT(name, 10), '...', RIGHT(name, 10))  FROM `book`; /*Вивести значення наступних колонок: перші 10 і останні 10 символів назви прописними буквами, розділені '...'*/

SELECT name, datee, DAY(datee) AS daydate, MONTH(datee) AS monthdate, YEAR(datee) AS yeardate FROM `book`; /*Вивести значення наступних колонок: назва, дата, день, місяць, рік*/
